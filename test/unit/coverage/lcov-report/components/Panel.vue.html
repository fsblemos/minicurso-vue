<template>
  <div class="panel is-marginless">
    <p class="panel-heading">
      {{ title }}
      <span class="icon" @click="toggle()">
        <i :class="[opened ? 'fa fa-angle-down' : 'fa fa-angle-up' ]"></i>
      </span>
    </p>
    <transition name="collapse">
      <div class="panel-block" v-show="opened">
        <slot></slot>
      </div>
    </transition>
  </div>
</template>
 
<script>
export default {
  name: 'panel',
  props: {
    title: String,
  },
  data() {
    return {
      opened: true,
    };
  },
  methods: {
    close() {
      this.opened = false;
    },
    toggle() {
      this.opened = !this.opened;
    },
  },
};
</script>
